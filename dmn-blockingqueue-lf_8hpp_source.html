436 DMN_PROC_CLEANUP_PUSH(&Dmn_BlockingQueue_Lf<T>::cleanup_thunk_inflight,

439 do {
440 auto data = popOptional(false, inflightTicket);
441 if (data) {
442 res.push_back(std::move_if_noexcept(*data));
443 } else {
444 dmn::Dmn_Proc::yield();
445 }
446 } while ((!isShutdown()) && res.size() < count &&
447 (0 == timeout || std::chrono::high_resolution_clock::now() < end));
448
449 DMN_PROC_CLEANUP_POP(0);

470template <typename T>
471auto Dmn_BlockingQueue_Lf<T>::popOptional(
472 bool wait, const Inflight_Guard_Ticket &inflightTicket)
473 -> std::optional<T> {
474 std::optional<T> res{};
475
476 while (true) {
477 Node *last = m_tail.load(std::memory_order_acquire);
478 if (nullptr == last) {
479 break;
480 }
481
482 Node *first = m_head.load(std::memory_order_acquire);
483 Node *next = first->m_next.load(std::memory_order_acquire);
484
485 if (first == m_head.load(std::memory_order_acquire)) {
486 if (first == last) {
487 if (next == nullptr) {
488 if (!wait) {
489 break;
490 }
491
492 while (last == m_tail.load(std::memory_order_acquire)) {
493 dmn::Dmn_Proc::testcancel();
494 dmn::Dmn_Proc::yield();
495 }
496
497 continue;
498 }
499
500 m_tail.compare_exchange_strong(
501 last, next, std::memory_order_release,
502 std::memory_order_relaxed); // Help move tail
503 } else {
504 if (m_head.compare_exchange_weak(first, next, std::memory_order_acq_rel,
505 std::memory_order_acquire)) {
506 res = std::move_if_noexcept(next->m_data);
507
508 retireNode(inflightTicket->getValue(), first);
509
510 break;
511 }
512 }
513 }
514 }
515
516 return res;
517}

541template <typename T>
542template <class U>
543void Dmn_BlockingQueue_Lf<T>::pushImpl(U &&item) {
544 Node *newNode = new Node;
545
546 newNode->m_data = std::forward<U>(item);
547
548 Node *last{};
549 Node *next{};
550
551 while (true) {
552 last = m_tail.load(std::memory_order_acquire);
553 if (nullptr == last) {
554 delete newNode;
555 newNode = nullptr;
556
557 throw std::runtime_error(
558 "Dmn_BlockingQueue_Lf: push attempted on shutdown queue");
559 }
560
561 next = last->m_next.load(std::memory_order_acquire);
562
563 if (last ==
564 m_tail.load(
565 std::memory_order_acquire)) { // Are tail and next consistent?
566 if (next == nullptr) {
567 // Release: ensures newNode->m_data is visible to any consumer that
568 // acquires this m_next pointer (publish-subscribe ordering).
569 if (last->m_next.compare_exchange_strong(next, newNode,
570 std::memory_order_release,
571 std::memory_order_relaxed)) {
572 break;
573 }
574 } else {
575 // Help advance the tail past a lagging pointer; release so that
576 // readers of m_tail see the node's data via the earlier m_next publish.
577 m_tail.compare_exchange_strong(last, next, std::memory_order_release,
578 std::memory_order_relaxed);
579 }
580 }
581 }
582
583 if (newNode) {
584 // Release: publishes the newly linked node so consumers see it via m_tail.
585 m_tail.compare_exchange_strong(last, newNode, std::memory_order_release,
586 std::memory_order_relaxed);
587 m_tail.notify_all();
588
589 // Relaxed: m_total_push_count is an exact counter (used by waitForEmpty());
590 // we only require atomicity for the increment, not additional ordering.
591 m_total_push_count.fetch_add(1, std::memory_order_relaxed);
592 }
593}

595template <typename T> auto Dmn_BlockingQueue_Lf<T>::waitForEmpty() -> uint64_t {
596 while (true) {
597 Node *last = m_tail.load(std::memory_order_acquire);
598 if (nullptr == last) {
599 break;
600 }
601
602 Node *first = m_head.load(std::memory_order_acquire);
603 Node *next = first->m_next.load(std::memory_order_acquire);
604
605 if (first == m_head.load(std::memory_order_acquire)) {
606 if (first == last) {
607 if (next == nullptr) {
608 break; // empty!
609 }
610 }
611 }
612
613 dmn::Dmn_Proc::testcancel();
614 dmn::Dmn_Proc::yield();
615 }
616
617 return m_total_push_count.load(std::memory_order_acquire);
618}

620template <typename T>
621void Dmn_BlockingQueue_Lf<T>::retireNode(uint64_t epochIndex, Node *node) {
622 assert(epochIndex < s_epochDataSize);
623
624 do {
625 auto first = m_epochReclaimNode[epochIndex].load(std::memory_order_acquire);
626
627 // Relaxed: the subsequent CAS (release) will order this store.
628 node->m_retired_next.store(first, std::memory_order_relaxed);
629
630 if (!m_epochReclaimNode[epochIndex].compare_exchange_strong(
631 first, node, std::memory_order_release,
632 std::memory_order_acquire)) {
633 continue;
634 }
635
636 break;
637 } while (true);
638}

640template <typename T> void Dmn_BlockingQueue_Lf<T>::shutdown() {
641 // 1. set shutdown flag
642 Dmn_BlockingQueue<Dmn_BlockingQueue_Lf<T>, T>::shutdown();
643
644 // 2. detach m_tail and mark data as empty.
645 // Release: establishes synchronization with threads that acquire m_tail in
646 // pushImpl/popOptional, ensuring they observe the shutdown state.
647 m_tail.store(nullptr, std::memory_order_release);
648 m_tail.notify_all();
649
650 // 3. wait for all threads that already "entered the epoch" to leave
651 this->waitForEmptyInflight();
652}

654template <typename T>
655auto Dmn_BlockingQueue_Lf<T>::enterInflightGuardFnc() -> uint64_t {
656 uint64_t epochIndex = 0;
657 auto total = m_in_flight_total.fetch_add(1, std::memory_order_acq_rel);
658
659 do {
660 auto ep = m_epochData.load(std::memory_order_acquire);
661 if ((total - ep.m_in_flight_total) >= s_epochTimeScale ||
662 total < ep.m_in_flight_total) {
663 EpochData epNew{.m_in_flight_total = total, .m_id = ep.m_id + 1};
664
665 if (!m_epochData.compare_exchange_strong(ep, epNew,
666 std::memory_order_release,
667 std::memory_order_acquire)) {
668 continue;
669 }
670
671 epochIndex = calc_epoch_index(epNew.m_id);
672
673 uint64_t index = 0;
674 while (index < s_epochDataSize) {
675 if (epochIndex != index) {
676 auto count =
677 m_epochInFlightCount[index].load(std::memory_order_acquire);
678
679 if (0 == count) {
680 auto ptr =
681 m_epochReclaimNode[index].load(std::memory_order_acquire);
682 if (nullptr != ptr &&
683 !m_epochReclaimNode[index].compare_exchange_strong(
684 ptr, nullptr, std::memory_order_release,
685 std::memory_order_acquire)) {
686 continue;
687 }
688
689 freeRetiredNodeList(ptr);
690 }
691 }
692
693 index++;
694 }
695 } else {
696 epochIndex = calc_epoch_index(ep.m_id);
697 }
698
699 // acq_rel: acquire prevents subsequent operations in this critical
700 // section from being reordered before we register this "in-flight"
701 // entry; release pairs with decrements on the same counter so that
702 // all work done under this guard happens-before leaving the epoch.
703 m_epochInFlightCount[epochIndex].fetch_add(1, std::memory_order_acq_rel);
704
705 break;
706 } while (true);
707
708 return epochIndex;
709}

716template <typename T>
717void Dmn_BlockingQueue_Lf<T>::leaveInflightGuardFnc(
718 const uint64_t &epochIndex) noexcept {
719 // acq_rel: acquire synchronizes with the registration fetch_add so that
720 // shared state is visible; release ensures all operations performed under
721 // this guard are visible to any thread that observes the count reaching
722 // 0.
723 if (1 == m_epochInFlightCount[epochIndex].fetch_sub(
724 1, std::memory_order_acq_rel)) {
725 auto ep = m_epochData.load(std::memory_order_acquire);
726 auto globalEpochIndex = calc_epoch_index(ep.m_id);
727
728 if (globalEpochIndex != epochIndex) {
729 do {
730 auto ptr =
731 m_epochReclaimNode[epochIndex].load(std::memory_order_acquire);
732 if (nullptr != ptr &&
733 !m_epochReclaimNode[epochIndex].compare_exchange_strong(
734 ptr, nullptr, std::memory_order_release,
735 std::memory_order_acquire)) {
736 continue;
737 }
738
739 freeRetiredNodeList(ptr);

741 break;
