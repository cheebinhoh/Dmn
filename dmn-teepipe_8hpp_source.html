313template <typename T> Dmn_TeePipe<T>::~Dmn_TeePipe() noexcept try {
314 // this is important as the conveyor thread uses the conditional variable
315 // and mutex, so we need to stop the thread before destroying both objects
316 // at the end of life of the Dmn_TeePipe object, as otherwise, those threads
317 // will fail to be awaken from the conditional variable state, and leak to
318 // system wide leak of available thread resource.
319 m_conveyor = {};

344template <typename T>
345void Dmn_TeePipe<T>::removeDmn_TeePipeSource(
346 std::shared_ptr<typename Dmn_TeePipe<T>::Dmn_TeePipeSource> &tps) {
347 assert(nullptr != tps);
348 assert(this == tps->m_teepipe);
349
350 std::unique_lock<std::mutex> lock(m_mutex);
351
352 Dmn_Proc::testcancel();
353
354 auto iter =
355 std::find_if(m_buffers.begin(), m_buffers.end(),
356 [tps](std::shared_ptr<Dmn_TeePipeSource> sp_iterTps) {
357 return tps.get() == sp_iterTps.get();
358 });
359
360 if (iter != m_buffers.end()) {
361 m_empty_cond.wait(lock, [this, &tps] { return tps->empty(); });
362 m_buffers.erase(iter);
363 tps = {};
364 }
365
366 lock.unlock();
367
368 m_cond.notify_all();
369}

377template <typename T>
378auto Dmn_TeePipe<T>::wait(bool no_open_source) -> uint64_t {
379 std::unique_lock<std::mutex> lock(m_mutex);
380
381 Dmn_Proc::testcancel();
382
383 // only returns if no other object owns the Dmn_TeePipeSource than
384 // Dmn_TeePipe
385 m_cond.wait(lock, [no_open_source, this] {
386 return !(no_open_source && (!m_buffers.empty()) &&
387 std::count_if(m_buffers.begin(), m_buffers.end(), [](auto &item) {
388 return item.use_count() > 1;
389 }) > 0);
390 });
391
392 // only returns if no data in buffer from Dmn_TeePipeSource to conveyor
393 m_empty_cond.wait(lock, [this] { return m_fill_buffer_count <= 0; });
394
395 return Dmn_Pipe<T>::waitForEmpty();
396}

402template <typename T> void Dmn_TeePipe<T>::runConveyorExec() {
403 m_conveyor->exec([this]() {
404 while (true) {
405 std::unique_lock<std::mutex> lock(m_mutex);
406
407 Dmn_Proc::testcancel();
408
409 m_cond.wait(lock, [this] {
410 return !(m_buffers.empty() || m_fill_buffer_count < m_buffers.size());
411 });
412
413 std::vector<T> post_processing_buffers{};
414
415 for (auto &tps : m_buffers) {
416 m_fill_buffer_count--;
417
418 auto data = tps->read();
419 assert(data);
420
421 post_processing_buffers.push_back(std::move_if_noexcept(*data));
422 }
423
424 if (m_post_processing_task_fn != nullptr) {
425 m_post_processing_task_fn(post_processing_buffers);
426 }
427
428 for (auto &data : post_processing_buffers) {
429 write(data);
430 }
431
432 post_processing_buffers.clear();
433
434 lock.unlock();
435
436 m_empty_cond.notify_all();
437 }
438 });
439}
