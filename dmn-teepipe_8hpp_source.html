292template <typename T> Dmn_TeePipe<T>::~Dmn_TeePipe() noexcept try {
293 // this is important as the conveyor thread uses the conditional variable
294 // and mutex, so we need to stop the thread before destroying both objects
295 // at the end of life of the Dmn_TeePipe object, as otherwise, those threads
296 // will fail to be awaken from the conditional variable state, and leak to
297 // system wide leak of available thread resource.

323template <typename T>
324void Dmn_TeePipe<T>::removeDmn_TeePipeSource(
325 std::shared_ptr<typename Dmn_TeePipe<T>::Dmn_TeePipeSource> &tps) {
326 assert(nullptr != tps);
327 assert(this == tps->m_teepipe);
328
329 std::unique_lock<std::mutex> lock(m_mutex);
330
331 Dmn_Proc::testcancel();
332
333 auto iter =
334 std::find_if(m_buffers.begin(), m_buffers.end(),
335 [tps](std::shared_ptr<Dmn_TeePipeSource> sp_iterTps) {
336 return tps.get() == sp_iterTps.get();
337 });
338
339 if (iter != m_buffers.end()) {
340 m_empty_cond.wait(lock, [this, &tps] { return tps->empty(); });
341 m_buffers.erase(iter);
342 tps = {};
343 }
344
345 lock.unlock();
346
347 m_cond.notify_all();
348}

356template <typename T>
357auto Dmn_TeePipe<T>::wait(bool no_open_source) -> uint64_t {
358 std::unique_lock<std::mutex> lock(m_mutex);
359
360 Dmn_Proc::testcancel();
361
362 // only returns if no other object owns the Dmn_TeePipeSource than
363 // Dmn_TeePipe
364 m_cond.wait(lock, [no_open_source, this] {
365 return !(no_open_source && (!m_buffers.empty()) &&
366 std::count_if(m_buffers.begin(), m_buffers.end(), [](auto &item) {
367 return item.use_count() > 1;
368 }) > 0);
369 });
370
371 // only returns if no data in buffer from Dmn_TeePipeSource to conveyor
372 m_empty_cond.wait(lock, [this] { return m_fill_buffer_count <= 0; });
373
374 return Dmn_Pipe<T>::waitForEmpty();
375}

381template <typename T> void Dmn_TeePipe<T>::runConveyorExec() {
382 m_conveyor->exec([this]() {
383 while (true) {
384 std::unique_lock<std::mutex> lock(m_mutex);
385
386 Dmn_Proc::testcancel();
387
388 m_cond.wait(lock, [this] {
389 return !(m_buffers.empty() || m_fill_buffer_count < m_buffers.size());
390 });
391
392 std::vector<T> post_processing_buffers{};
393
394 for (auto &tps : m_buffers) {
395 m_fill_buffer_count--;
396
397 auto data = tps->read();
398 assert(data);
399
400 post_processing_buffers.push_back(std::move_if_noexcept(*data));
401 }
402
403 if (m_post_processing_task_fn != nullptr) {
404 m_post_processing_task_fn(post_processing_buffers);
405 }
406
407 for (auto &data : post_processing_buffers) {
408 write(data);
409 }
410
411 post_processing_buffers.clear();
412
413 lock.unlock();
414
415 m_empty_cond.notify_all();
416 }
417 });
418}
