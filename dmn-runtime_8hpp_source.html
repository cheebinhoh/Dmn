627template <template <class> class QueueType>
628void Dmn_Runtime_Manager<QueueType>::execRuntimeJobInternal() {
629 Dmn_Runtime_Job::Priority state{}; // not high, not medium, not low
630
631 assert(isRunInAsyncThread());
632 assert(this->m_sched_job.size() == this->m_sched_task.size());
633
634 if (!this->m_sched_job.empty()) {
635 state = this->m_sched_job.top().m_priority;
636 }
637
638 try {
639 if (state != Dmn_Runtime_Job::Priority::kHigh) {
640 auto item = m_highQueue.popNoWait();
641
642 if (item) {
643 this->addRuntimeJobToCoroutineSchedulerContext(std::move(*item));
644 } else if (state != Dmn_Runtime_Job::Priority::kMedium) {
645 item = m_mediumQueue.popNoWait();
646
647 if (item) {
648 this->addRuntimeJobToCoroutineSchedulerContext(std::move(*item));
649 } else if (state != Dmn_Runtime_Job::Priority::kLow) {
650
651 item = m_lowQueue.popNoWait();
652 if (item) {
653 this->addRuntimeJobToCoroutineSchedulerContext(std::move(*item));
654 }
655 }
656 }
657 }
658
659 assert(this->m_sched_job.size() == this->m_sched_task.size());
660
661 auto jobsCountInScheduler = this->m_sched_job.size();
662
663 if (jobsCountInScheduler > 0) {
664 bool done = this->runRuntimeCoroutineScheduler();
665
666 if (!done) {
667 assert(!this->m_sched_job.empty());
668 assert(!this->m_sched_task.empty());
669 assert(this->m_sched_job.size() == jobsCountInScheduler);
670
671 this->runRuntimeJobExecutor();
672 } else if (this->m_sched_job.size() < jobsCountInScheduler &&
673 this->m_jobs_count.fetch_sub(1) > 1) {
674 this->runRuntimeJobExecutor();
675 } else {
676 assert(this->m_sched_job.empty());
677 assert(this->m_sched_task.empty());
678 assert(0 == m_jobs_count.load(std::memory_order_acquire));
679 }
680 }
681 } catch (...) {
682 if (this->m_jobs_count.fetch_sub(1) > 1) {
683 this->runRuntimeJobExecutor();
684 }
685 }
686}

692template <template <class> class QueueType>
693void Dmn_Runtime_Manager<QueueType>::exitMainLoop() {
694 if (!m_main_exit_atomic_flag.test_and_set(std::memory_order_release)) {
695 m_main_enter_atomic_flag.clear(std::memory_order_release);
696 m_main_exit_atomic_flag.notify_all();
697
698 // set the last timer, so that signalWaitProc gracefully exit.
699 this->setNextTimer(0, 10000);
700
701 // after setting the main loop enter (false) and exit (true) flags, we just
702 // have to wait for signalWaitProc to gracefully exit as it is triggered
703 // at fixed intervals before we clear the timer.
704 if (m_signalWaitProc) {
705 m_signalWaitProc->wait();
706 m_signalWaitProc = {};
707 }
708 }
709
710 if (m_pimpl) {
711 this->setNextTimer(
712 0, 0); // disable timer, though m_signalWaitProc should already have
713 // done so; if it has crashed, we still disable the timer here.
714 }

722template <template <class> class QueueType>
723void Dmn_Runtime_Manager<QueueType>::enterMainLoop() {
724 // If there are already pending jobs (e.g., from a prior run), ensure the
725 // runtime job executor is started so the system does not remain idle.
726 bool startJobExecutor = m_jobs_count.load(std::memory_order_acquire) > 0;
727
728 if (m_main_enter_atomic_flag.test_and_set(std::memory_order_acquire)) {
729 assert(false && "Error: enter main loop twice without exit");
730
731 throw std::runtime_error("Error: enter main loop twice without exit");
732 }
733
734 m_main_exit_atomic_flag.clear(std::memory_order_release);
735 m_main_enter_atomic_flag.notify_all();
736 Dmn_Proc::yield();
737
738 m_signalWaitProc = std::make_unique<Dmn_Proc>(
739 "DmnRuntimeManager_SignalWait", [this]() -> void {
740 while (true) {
741 int signo{};
742 int err{};
743
744 err = sigwait(&m_mask, &signo);
745 if (err) {
746 throw std::runtime_error("Error in sigwait: " +
747 std::system_category().message(err));
748 }
749
750 if (m_main_exit_atomic_flag.test(std::memory_order_acquire)) {
751 break;
752 } else {
753 this->addExecTask([this, signo]() {
754 this->execSignalHandlerHookInternal(signo);
755 });
756 }
757 }
758 });
759
760 if (!m_signalWaitProc->exec()) {
761 throw std::runtime_error(
762 "Failed to start DmnRuntimeManager_SignalWait task");
763 }
764
765 if (startJobExecutor) {
766 this->runRuntimeJobExecutor();
767 }
768
769 while (!m_main_exit_atomic_flag.test(std::memory_order_acquire)) {
770 m_main_exit_atomic_flag.wait(false, std::memory_order_acquire);
771 }
772}

840 // Mask signals before creating any threads so they are blocked
841 // in all subsequently created threads (avoids races).
842 sigset_t old_mask{};
843 int err{};
844
845 sigemptyset(&Dmn_Runtime_Manager::s_mask);

871template <template <class> class QueueType>
872auto Dmn_Runtime_Manager<QueueType>::runRuntimeCoroutineScheduler() -> bool {
873 bool complete{true};
874
875 assert(isRunInAsyncThread());
876 assert(this->m_sched_task.size() > 0);
877 assert(this->m_sched_job.size() > 0);
878 assert(this->m_sched_task.size() == this->m_sched_job.size());
879
880 const Dmn_Runtime_Job &runningJob = this->m_sched_job.top();
881
882 auto &task = this->m_sched_task.top();
883
884 try {
885 if (task.isValid()) {
886 // runtime jobs must complete before returning to scheduler; suspension is
887 // only used for intra-job yielding and will be resumed immediately.
888 do {
889 task.m_handle.resume();
890
891 if (task.m_handle.done()) {
892 auto &ep = task.m_handle.promise().m_except;
893 if (ep && runningJob.m_onErrorFnc) {
894 std::rethrow_exception(ep);
895 }
896
897 break;
898 } else {
899 assert(!this->m_sched_job.empty());
900 assert(!this->m_sched_task.empty());
901
902 complete = false;
903 break;
904 }
905 } while (true);
906 }
907 } catch (...) {
908 try {
909 if (runningJob.m_onErrorFnc) {
910 std::exception_ptr ep = std::current_exception();
911
912 runningJob.m_onErrorFnc(ep);
913 }
914 } catch (...) {
915 // ignore it.
916 }
917 }
918
919 if (complete) {
920 this->m_sched_task.pop();
921 this->m_sched_job.pop();
922 }
923
924 return complete;
925}
