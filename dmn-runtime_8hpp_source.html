666template <template <class> class QueueType>
667void Dmn_Runtime_Manager<QueueType>::execRuntimeJobInternal() {
668 Dmn_Runtime_Job::Priority state{}; // not high, not medium, not low
669
670 assert(isRunInAsyncThread());
671 assert(this->m_sched_job.size() == this->m_sched_task.size());
672
673 if (!this->m_sched_job.empty()) {
674 state = this->m_sched_job.top().m_priority;
675 }
676
677 try {
678 if (state != Dmn_Runtime_Job::Priority::kHigh) {
679 auto item = m_highQueue.popNoWait();
680
681 if (item) {
682 this->addRuntimeJobToCoroutineSchedulerContext(std::move(*item));
683 } else if (state != Dmn_Runtime_Job::Priority::kMedium) {
684 item = m_mediumQueue.popNoWait();
685
686 if (item) {
687 this->addRuntimeJobToCoroutineSchedulerContext(std::move(*item));
688 } else if (state != Dmn_Runtime_Job::Priority::kLow) {
689
690 item = m_lowQueue.popNoWait();
691 if (item) {
692 this->addRuntimeJobToCoroutineSchedulerContext(std::move(*item));
693 }
694 }
695 }
696 }
697
698 assert(this->m_sched_job.size() == this->m_sched_task.size());
699
700 auto jobsCountInScheduler = this->m_sched_job.size();
701
702 if (jobsCountInScheduler > 0) {
703 bool done = this->runRuntimeCoroutineScheduler();
704
705 if (!done) {
706 assert(!this->m_sched_job.empty());
707 assert(!this->m_sched_task.empty());
708 assert(this->m_sched_job.size() == jobsCountInScheduler);
709
710 this->runRuntimeJobExecutor();
711 } else if (this->m_sched_job.size() < jobsCountInScheduler &&
712 this->m_jobs_count.fetch_sub(1) > 1) {
713 this->runRuntimeJobExecutor();
714 } else {
715 assert(this->m_sched_job.empty());
716 assert(this->m_sched_task.empty());
717 assert(0 == m_jobs_count.load(std::memory_order_acquire));
718 }
719 }
720 } catch (...) {
721 if (this->m_jobs_count.fetch_sub(1) > 1) {
722 this->runRuntimeJobExecutor();
723 }
724 }
725}

731template <template <class> class QueueType>
732void Dmn_Runtime_Manager<QueueType>::exitMainLoop() {
733 if (!m_main_exit_flag.test_and_set(std::memory_order_release)) {
734 m_main_enter_flag.clear(std::memory_order_release);
735
736 // set the last timer, so that signalWaitProc gracefully exit.
737 this->setNextTimer(0, 10000);
738
739 // after setting the main loop enter (false) and exit (true) flags, we just
740 // have to wait for signalWaitProc to gracefully exit as it is triggered
741 // at fixed intervals before we clear the timer.
742 if (m_signalWaitProc) {
743 m_signalWaitProc->wait();
744 m_signalWaitProc = {};
745 }
746 }
747
748 if (m_pimpl) {
749 this->setNextTimer(
750 0, 0); // disable timer, though m_signalWaitProc should already have
751 // done so; if it has crashed, we still disable the timer here.
752 }
753
754 m_main_exit_flag.notify_all();
755}

765template <template <class> class QueueType>
766void Dmn_Runtime_Manager<QueueType>::enterMainLoop() {
767 // If there are already pending jobs (e.g., from a prior run), ensure the
768 // runtime job executor is started so the system does not remain idle.
769 bool startJobExecutor = m_jobs_count.load(std::memory_order_acquire) > 0;
770
771 if (m_main_enter_flag.test_and_set(std::memory_order_acquire)) {
772 assert(false && "Error: enter main loop twice without exit");
773
774 throw std::runtime_error("Error: enter main loop twice without exit");
775 }
776
777 m_main_exit_flag.clear(std::memory_order_release);
778 m_main_enter_flag.notify_all();
779 Dmn_Proc::yield();
780
781 m_signalWaitProc = std::make_unique<Dmn_Proc>(
782 "DmnRuntimeManager_SignalWait", [this]() -> void {
783 while (true) {
784 int signo{};
785 int err{};
786
787 err = sigwait(&m_mask, &signo);
788 if (err) {
789 throw std::runtime_error("Error in sigwait: " +
790 std::system_category().message(err));
791 }
792
793 if (m_main_exit_flag.test(std::memory_order_acquire)) {
794 break;
795 } else {
796 this->addExecTask([this, signo]() {
797 this->execSignalHandlerHookInternal(signo);
798 });
799 }
800 }
801 });
802
803 if (!m_signalWaitProc->exec()) {
804 throw std::runtime_error(
805 "Failed to start DmnRuntimeManager_SignalWait task");
806 }
807
808 if (startJobExecutor) {
809 this->runRuntimeJobExecutor();
810 }
811
812 while (!m_main_exit_flag.test(std::memory_order_acquire)) {
813 m_main_exit_flag.wait(false, std::memory_order_acquire);
814 }
815}

882template <template <class> class QueueType>
883void Dmn_Runtime_Manager<QueueType>::runPriorToCreateInstance() {
884 // Mask signals before creating any threads so they are blocked
885 // in all subsequently created threads (avoids races).
886 sigset_t old_mask{};
887 int err{};
888
889 sigemptyset(&Dmn_Runtime_Manager::s_mask);
890 sigaddset(&Dmn_Runtime_Manager::s_mask, SIGALRM);
891 sigaddset(&Dmn_Runtime_Manager::s_mask, SIGINT);
892 sigaddset(&Dmn_Runtime_Manager::s_mask, SIGTERM);
893 sigaddset(&Dmn_Runtime_Manager::s_mask, SIGQUIT);
894 sigaddset(&Dmn_Runtime_Manager::s_mask, SIGHUP);
895
896 err = pthread_sigmask(SIG_BLOCK, &Dmn_Runtime_Manager::s_mask, &old_mask);
897 if (0 != err) {
898 throw std::runtime_error("Error in pthread_sigmask: " +
899 std::string{strerror(err)});

915template <template <class> class QueueType>
916auto Dmn_Runtime_Manager<QueueType>::runRuntimeCoroutineScheduler() -> bool {
917 bool complete{true};
918
919 assert(isRunInAsyncThread());
920 assert(this->m_sched_task.size() > 0);
921 assert(this->m_sched_job.size() > 0);
922 assert(this->m_sched_task.size() == this->m_sched_job.size());
923
924 const Dmn_Runtime_Job &runningJob = this->m_sched_job.top();
925
926 auto &task = this->m_sched_task.top();
927
928 try {
929 if (task.isValid()) {
930 // runtime jobs must complete before returning to scheduler; suspension is
931 // only used for intra-job yielding and will be resumed immediately.
932 do {
933 task.m_handle.resume();
934
935 if (task.m_handle.done()) {
936 auto &ep = task.m_handle.promise().m_except;
937 if (ep && runningJob.m_onErrorFnc) {
938 std::rethrow_exception(ep);
939 }
940
941 break;
942 } else {
943 assert(!this->m_sched_job.empty());
944 assert(!this->m_sched_task.empty());
945
946 complete = false;
947 break;
948 }
949 } while (true);
950 }
951 } catch (...) {
952 try {
953 if (runningJob.m_onErrorFnc) {
954 std::exception_ptr ep = std::current_exception();
955
956 runningJob.m_onErrorFnc(ep);
957 }
958 } catch (...) {
959 // ignore it.
960 }
961 }
962
963 if (complete) {
964 this->m_sched_task.pop();
965 this->m_sched_job.pop();
966 }
967
968 return complete;
969}
