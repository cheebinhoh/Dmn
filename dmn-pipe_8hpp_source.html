267template <typename T, typename QueueType>
268auto Dmn_Pipe<T, QueueType>::readAndProcess(Dmn_Pipe::Task fn, size_t count,
269 long timeout) -> size_t {
270 auto dataList = this->pop(count, timeout);
271
272 size_t processedCount = dataList.size();
273
274 if (fn) {
275 for (auto &item : dataList) {
276 Dmn_Proc::testcancel();
277
278 fn(std::move_if_noexcept(item));
279 }
280 }
281
282 std::unique_lock<std::mutex> lock(m_mutex);
283
284 m_count += processedCount;
285
286 m_empty_cond.notify_all();
287
288 return processedCount;
289}
