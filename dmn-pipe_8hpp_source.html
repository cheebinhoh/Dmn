268template <typename T, typename QueueType>
269auto Dmn_Pipe<T, QueueType>::readAndProcess(Dmn_Pipe::Task fn, size_t count,
270 long timeout) -> size_t {
271 auto dataList = this->pop(count, timeout);
272
273 size_t processedCount = dataList.size();
274
275 if (fn) {
276 for (auto &item : dataList) {
277 Dmn_Proc::testcancel();
278
279 fn(std::move_if_noexcept(item));
280 }
281 }
282
283 std::unique_lock<std::mutex> lock(m_mutex);
284
285 m_count += processedCount;
286
287 m_empty_cond.notify_all();
288
289 return processedCount;
290}
