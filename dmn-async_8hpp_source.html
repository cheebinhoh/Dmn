234template <template <class> class QueueType>
235Dmn_Async<QueueType>::Dmn_Async(std::string_view name) : m_name{name} {
236 m_pipe = std::make_unique<BasePipe>(
237 m_name,
238 [this](std::shared_ptr<Dmn_Async::Dmn_Async_Handle> task) -> void {
239 try {
240 if (task->m_due_in_future > 0) {
241 const long long now =
242 std::chrono::duration_cast<std::chrono::nanoseconds>(
243 std::chrono::steady_clock::now().time_since_epoch())
244 .count();
245
246 if (now >= task->m_due_in_future) {
247 if (task->m_fnc) {
248 task->m_fnc();
249 }
250
251 task->m_p.set_value();
252 } else {
253 this->m_pipe->write(task);
254 }
255 } else {
256 if (task->m_fnc) {
257 task->m_fnc();
258 }
259
260 task->m_p.set_value();
261 }
262 } catch (...) {
263 task->m_p.set_exception(std::current_exception());
264 }
265
266 Dmn_Proc::yield();
267 });
268}
