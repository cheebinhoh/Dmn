57// A simple rendezvous object returned to callers that want to wait for a
58// previously submitted asynchronous task to finish. Calling wait() blocks
59// until the task has completed. If the task threw, the stored exception
60// will be rethrown to the waiter.

183 long long time_in_future =
184 std::chrono::duration_cast<std::chrono::nanoseconds>(
185 std::chrono::steady_clock::now().time_since_epoch())
186 .count() +
187 std::chrono::duration_cast<std::chrono::nanoseconds>(duration).count();
188
189 auto task_shared_ptr =
190 std::make_shared<Dmn_Async_Handle>(fnc, time_in_future);

193 this->write(task_shared_ptr);

198template <template <class> class QueueType>
199Dmn_Async<QueueType>::Dmn_Async(std::string_view name)
200 : BasePipe{
201 name, [this](std::shared_ptr<Dmn_Async_Handle> task) -> void {
202 try {
203 if (task->m_due_in_future > 0) {
204 const long long now =
205 std::chrono::duration_cast<std::chrono::nanoseconds>(
206 std::chrono::steady_clock::now().time_since_epoch())
207 .count();
208
209 if (now >= task->m_due_in_future) {
210 if (task->m_fnc) {
211 task->m_fnc();
212 }
213
214 task->m_p.set_value();
215 } else {
216 this->write(task);
217 }
218 } else {
219 if (task->m_fnc) {
220 task->m_fnc();
221 }
222
223 task->m_p.set_value();
224 }
225 } catch (...) {
226 task->m_p.set_exception(std::current_exception());
227 }
228
229 Dmn_Proc::yield();
230 }} {}
