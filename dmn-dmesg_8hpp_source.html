652template <class... U> auto Dmn_DMesg::openHandler(U &&...arg) -> HandlerType {
653 // This function:
654 // - constructs a handler
655 // - registers the handler as a subscriber
656 // - wires handler<->subscriber<->publisher links
657 // - schedules an async task on the publisher's singleton async thread to:
658 // * add the handler to the internal list
659 // * playback last-known messages per topic
660 // * mark handler as initialized after playback
661 //
662 // The use of the publisher's singleton async context keeps most operations
663 // mutex-free for the hot paths (publish/notify).
664
665 auto handlerProxy = Dmn_DMesg::Dmn_DMesgHandlerProxy();
666
667 std::shared_ptr<Dmn_DMesg::Dmn_DMesgHandler> handler =
668 std::make_shared<Dmn_DMesg::Dmn_DMesgHandler>(std::forward<U>(arg)...);
669
670 handlerProxy.m_handler = handler;
671
672 /* The topic filter is executed within the DMesg singleton asynchronous
673 * thread context, but the filter value is maintained per Dmn_DMesgHandler.
674 * This design keeps DMesg itself mutex-free while remaining thread safe.
675 */
676 handler->m_owner = this;
677 handler->m_sub =
678 this->registerSubscriber<Dmn_DMesgHandler::Dmn_DMesgHandlerSub>();
679 handler->m_sub->m_owner = handler.get();
680
681 DMN_ASYNC_CALL_WITH_CAPTURE(
682 {
683 this->m_handlers.push_back(handler);
684 this->playbackLastTopicDMesgPbInternal();
685 handler->setAfterInitialPlayback();
686 },
687 this, handler);
688
689 return handlerProxy;
690}
