707template <class... U> auto Dmn_DMesg::openHandler(U &&...arg) -> HandlerType {
708 // This function:
709 // - constructs a handler
710 // - registers the handler as a subscriber
711 // - wires handler<->subscriber<->publisher links
712 // - schedules an async task on the publisher's singleton async thread to:
713 // * add the handler to the internal list
714 // * playback last-known messages per topic
715 // * mark handler as initialized after playback
716 //
717 // The use of the publisher's singleton async context keeps most operations
718 // mutex-free for the hot paths (publish/notify).
719
720 auto handlerProxy = Dmn_DMesg::Dmn_DMesgHandlerProxy();
721
722 std::shared_ptr<Dmn_DMesg::Dmn_DMesgHandler> handler =
723 std::make_shared<Dmn_DMesg::Dmn_DMesgHandler>(std::forward<U>(arg)...);
724
725 handlerProxy.m_handler = handler;
726
727 /* The topic filter is executed within the DMesg singleton asynchronous
728 * thread context, but the filter value is maintained per Dmn_DMesgHandler.
729 * This design keeps DMesg itself mutex-free while remaining thread safe.
730 */
731 handler->m_owner = this;
732 handler->m_sub =
733 this->registerSubscriber<Dmn_DMesgHandler::Dmn_DMesgHandlerSub>();
734 handler->m_sub->m_owner = handler.get();
735
736 auto waitHandler = this->addExecTaskWithWait([this, &handler]() {
737 this->m_handlers.push_back(handler);
738 this->playbackLastTopicDMesgPbInternal();
739 handler->setAfterInitialPlayback();
740 });
741
742 waitHandler->wait();
743
744 return handlerProxy;
745}
