695template <class... U> auto Dmn_DMesg::openHandler(U &&...arg) -> HandlerType {
696 // This function:
697 // - constructs a handler
698 // - registers the handler as a subscriber
699 // - wires handler<->subscriber<->publisher links
700 // - schedules an async task on the publisher's singleton async thread to:
701 // * add the handler to the internal list
702 // * playback last-known messages per topic
703 // * mark handler as initialized after playback
704 //
705 // The use of the publisher's singleton async context keeps most operations
706 // mutex-free for the hot paths (publish/notify).
707
708 auto handlerProxy = Dmn_DMesg::Dmn_DMesgHandlerProxy();
709
710 std::shared_ptr<Dmn_DMesg::Dmn_DMesgHandler> handler =
711 std::make_shared<Dmn_DMesg::Dmn_DMesgHandler>(std::forward<U>(arg)...);
712
713 handlerProxy.m_handler = handler;
714
715 /* The topic filter is executed within the DMesg singleton asynchronous
716 * thread context, but the filter value is maintained per Dmn_DMesgHandler.
717 * This design keeps DMesg itself mutex-free while remaining thread safe.
718 */
719 handler->m_owner = this;
720 handler->m_sub =
721 this->registerSubscriber<Dmn_DMesgHandler::Dmn_DMesgHandlerSub>();
722 handler->m_sub->m_owner = handler.get();
723
724 auto waitHandler = this->addExecTaskWithWait([this, &handler]() {
725 this->m_handlers.push_back(handler);
726 this->playbackLastTopicDMesgPbInternal();
727 handler->setAfterInitialPlayback();
728 });
729
730 waitHandler->wait();
731
732 return handlerProxy;
733}
