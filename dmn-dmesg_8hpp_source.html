681template <class... U> auto Dmn_DMesg::openHandler(U &&...arg) -> HandlerType {
682 // This function:
683 // - constructs a handler
684 // - registers the handler as a subscriber
685 // - wires handler<->subscriber<->publisher links
686 // - schedules an async task on the publisher's singleton async thread to:
687 // * add the handler to the internal list
688 // * playback last-known messages per topic
689 // * mark handler as initialized after playback
690 //
691 // The use of the publisher's singleton async context keeps most operations
692 // mutex-free for the hot paths (publish/notify).
693
694 auto handlerProxy = Dmn_DMesg::Dmn_DMesgHandlerProxy();
695
696 std::shared_ptr<Dmn_DMesg::Dmn_DMesgHandler> handler =
697 std::make_shared<Dmn_DMesg::Dmn_DMesgHandler>(std::forward<U>(arg)...);
698
699 handlerProxy.m_handler = handler;
700
701 /* The topic filter is executed within the DMesg singleton asynchronous
702 * thread context, but the filter value is maintained per Dmn_DMesgHandler.
703 * This design keeps DMesg itself mutex-free while remaining thread safe.
704 */
705 handler->m_owner = this;
706 handler->m_sub =
707 this->registerSubscriber<Dmn_DMesgHandler::Dmn_DMesgHandlerSub>();
708 handler->m_sub->m_owner = handler.get();
709
710 auto waitHandler = this->addExecTaskWithWait([this, &handler]() {
711 this->m_handlers.push_back(handler);
712 this->playbackLastTopicDMesgPbInternal();
713 handler->setAfterInitialPlayback();
714 });
715
716 waitHandler->wait();
717
718 return handlerProxy;
719}
