682template <class... U> auto Dmn_DMesg::openHandler(U &&...arg) -> HandlerType {
683 // This function:
684 // - constructs a handler
685 // - registers the handler as a subscriber
686 // - wires handler<->subscriber<->publisher links
687 // - schedules an async task on the publisher's singleton async thread to:
688 // * add the handler to the internal list
689 // * playback last-known messages per topic
690 // * mark handler as initialized after playback
691 //
692 // The use of the publisher's singleton async context keeps most operations
693 // mutex-free for the hot paths (publish/notify).
694
695 auto handlerProxy = Dmn_DMesg::Dmn_DMesgHandlerProxy();
696
697 std::shared_ptr<Dmn_DMesg::Dmn_DMesgHandler> handler =
698 std::make_shared<Dmn_DMesg::Dmn_DMesgHandler>(std::forward<U>(arg)...);
699
700 handler->m_owner = this;
701
702 this->registerSubscriber(handler);
703
704 handlerProxy.m_handler = handler;
705
706 /* The topic filter is executed within the DMesg singleton asynchronous
707 * thread context, but the filter value is maintained per Dmn_DMesgHandler.
708 * This design keeps DMesg itself mutex-free while remaining thread safe.
709 */
710 auto waitHandler = this->addExecTaskWithWait([this, &handler]() {
711 this->m_handlers.push_back(handler);
712 this->playbackLastTopicDMesgPbInternal();
713 handler->setAfterInitialPlayback();
714 });
715
716 waitHandler->wait();
717
718 return handlerProxy;
719}
