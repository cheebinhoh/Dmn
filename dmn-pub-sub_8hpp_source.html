354template <typename T, template <class> class QueueType>
355void Dmn_Pub<T, QueueType>::publishInternal(const T &item) {
356 /* Though through Dmn_Async (parent class), we have a mean to
357 * guarantee that only one thread is executing the core logic
358 * and manipulate the m_subscribers state in a singleton asynchronous
359 * thread context and without the use of mutex to protect the m_subscribers,
360 * but that also requires both registerSubscriber() and
361 * unregisterSubscriber() API methods to have it side effects executed
362 * in the singleton asynchronous thread context, and which means that upon
363 * returning from both API methods, the client is not guaranteed that
364 * it has registered or unregistered successfully.
365 *
366 * A simple mutex will guarantee that and the penalty of mutex is small
367 * as most clients are registered upon application brought up, and only
368 * deregister upon application shutdown.
369 */
370
371 std::unique_lock<std::mutex> lock(m_mutex);
372
373 /* Keep the published item in circular ring buffer for
374 * efficient access to playback to new subscribers whose misses the
375 * data.
376 */
377
378 m_buffer.push_back(item);
379
380 std::size_t numOfElementToBeRemoved = 0;
381 if (m_buffer.size() > m_capacity) {
382 numOfElementToBeRemoved = m_buffer.size() - m_capacity;
383 }
384
385 while (numOfElementToBeRemoved > 0 && !m_buffer.empty()) {
386 m_buffer.erase(m_buffer.begin());
387 numOfElementToBeRemoved--;
388 }
389
390 for (auto &sub : m_subscribers) {
391 if (!m_filter_fn || m_filter_fn(sub.get(), item)) {
392 sub->notifyInternal(item);
393 }
394 }
395} // method publishInternal()

397template <typename T, template <class> class QueueType>
398void Dmn_Pub<T, QueueType>::registerSubscriber(std::shared_ptr<Dmn_Sub> sub) {
399 std::unique_lock<std::mutex> lock(m_mutex);
400
401 if (this == sub->m_pub) {
402 return;
403 }
404
405 assert(nullptr == sub->m_pub ||
406 "The subscriber has been registered with another publisher");
407
408 sub->m_pub = this;
409 m_subscribers.push_back(sub);
410
411 // resend the data items that the registered subscriber
412 // miss.
413 size_t numberOfItemsToBeSkipped = 0;
414 if (sub->m_replayQuantity > 0 &&
415 m_buffer.size() > static_cast<size_t>(sub->m_replayQuantity)) {
416 numberOfItemsToBeSkipped =
417 m_buffer.size() - static_cast<size_t>(sub->m_replayQuantity);
418 }
419
420 for (auto &item : m_buffer) {
421 if (numberOfItemsToBeSkipped > 0) {
422 numberOfItemsToBeSkipped--;
423 } else {
424 sub->notifyInternal(item);
425 }
426 }
427}
