346template <typename T, template <class> class QueueType>
347void Dmn_Pub<T, QueueType>::publishInternal(const T &item) {
348 /* Though through Dmn_Async (parent class), we have a mean to
349 * guarantee that only one thread is executing the core logic
350 * and manipulate the m_subscribers state in a singleton asynchronous
351 * thread context and without the use of mutex to protect the m_subscribers,
352 * but that also requires both registerSubscriber() and
353 * unregisterSubscriber() API methods to have it side effects executed
354 * in the singleton asynchronous thread context, and which means that upon
355 * returning from both API methods, the client is not guaranteed that
356 * it has registered or unregistered successfully.
357 *
358 * A simple mutex will guarantee that and the penalty of mutex is small
359 * as most clients are registered upon application brought up, and only
360 * deregister upon application shutdown.
361 */
362
363 std::unique_lock<std::mutex> lock(m_mutex);
364
365 /* Keep the published item in circular ring buffer for
366 * efficient access to playback to new subscribers whose misses the
367 * data.
368 */
369
370 m_buffer.push_back(item);
371
372 std::size_t numOfElementToBeRemoved = 0;
373 if (m_buffer.size() > m_capacity) {
374 numOfElementToBeRemoved = m_buffer.size() - m_capacity;
375 }
376
377 while (numOfElementToBeRemoved > 0 && !m_buffer.empty()) {
378 m_buffer.erase(m_buffer.begin());
379 numOfElementToBeRemoved--;
380 }
381
382 for (auto &sub : m_subscribers) {
383 if (!m_filter_fn || m_filter_fn(sub.get(), item)) {
384 sub->notifyInternal(item);
385 }
386 }
387} // method publishInternal()

389template <typename T, template <class> class QueueType>
390void Dmn_Pub<T, QueueType>::registerSubscriber(std::shared_ptr<Dmn_Sub> sub) {
391 std::unique_lock<std::mutex> lock(m_mutex);
392
393 if (this == sub->m_pub) {
394 return;
395 }
396
397 assert(nullptr == sub->m_pub ||
398 "The subscriber has been registered with another publisher");
399
400 sub->m_pub = this;
401 m_subscribers.push_back(sub);
402
403 // resend the data items that the registered subscriber
404 // miss.
405 size_t numberOfItemsToBeSkipped = 0;
406 if (sub->m_replayQuantity > 0 &&
407 m_buffer.size() > static_cast<size_t>(sub->m_replayQuantity)) {
408 numberOfItemsToBeSkipped =
409 m_buffer.size() - static_cast<size_t>(sub->m_replayQuantity);
410 }
411
412 for (auto &item : m_buffer) {
413 if (numberOfItemsToBeSkipped > 0) {
414 numberOfItemsToBeSkipped--;
415 } else {
416 sub->notifyInternal(item);
417 }
418 }
419}
