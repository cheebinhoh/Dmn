305template <typename T, template <class> class QueueType>
306void Dmn_Pub<T, QueueType>::publishInternal(const T &item) {
307 /* Though through Dmn_Async (parent class), we have a mean to
308 * guarantee that only one thread is executing the core logic
309 * and manipulate the m_subscribers state in a singleton asynchronous
310 * thread context and without the use of mutex to protect the m_subscribers,
311 * but that also requires both registerSubscriber() and
312 * unregisterSubscriber() API methods to have it side effects executed
313 * in the singleton asynchronous thread context, and which means that upon
314 * returning from both API methods, the client is not guaranteed that
315 * it has registered or unregistered successfully.
316 */
317
318 /* Keep the published item in circular ring buffer for
319 * efficient access to playback to new subscribers whose misses the
320 * data.
321 */
322
323 m_buffer.push_back(item);
324
325 std::size_t numOfElementToBeRemoved = 0;
326 if (m_buffer.size() > m_capacity) {
327 numOfElementToBeRemoved = m_buffer.size() - m_capacity;
328 }
329
330 while (numOfElementToBeRemoved > 0 && !m_buffer.empty()) {
331 m_buffer.erase(m_buffer.begin());
332 numOfElementToBeRemoved--;
333 }
334
335 for (auto &sub : m_subscribers) {
336 if (!m_filter_fn || m_filter_fn(sub.get(), item)) {
337 sub->notify(item);
338 }
339 }
340} // method publishInternal()

342template <typename T, template <class> class QueueType>
343void Dmn_Pub<T, QueueType>::registerSubscriber(std::shared_ptr<Dmn_Sub> sub) {
344 auto waitHandler = this->addExecTaskWithWait([this, sub]() -> void {
345 if (this == sub->m_pub) {
346 return;
347 }
348
349 assert(nullptr == sub->m_pub ||
350 "The subscriber has been registered with another publisher");
351
352 sub->m_pub = this;
353 m_subscribers.push_back(sub);
354
355 // resend the data items that the registered subscriber
356 // miss.
357 size_t numberOfItemsToBeSkipped = 0;
358 if (sub->m_replayQuantity > 0 &&
359 m_buffer.size() > static_cast<size_t>(sub->m_replayQuantity)) {
360 numberOfItemsToBeSkipped =
361 m_buffer.size() - static_cast<size_t>(sub->m_replayQuantity);
362 }
363
364 for (auto &item : m_buffer) {
365 if (numberOfItemsToBeSkipped > 0) {
366 numberOfItemsToBeSkipped--;
367 } else {
368 sub->notify(item);
369 }
370 }
371 });
372
373 waitHandler->wait();
374}
