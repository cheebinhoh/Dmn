312template <typename T, template <class> class QueueType>
313void Dmn_Pub<T, QueueType>::publishInternal(const T &item) {
314 /* Though through Dmn_Async (parent class), we have a mean to
315 * guarantee that only one thread is executing the core logic
316 * and manipulate the m_subscribers state in a singleton asynchronous
317 * thread context and without the use of mutex to protect the m_subscribers,
318 * but that also requires both registerSubscriber() and
319 * unregisterSubscriber() API methods to have it side effects executed
320 * in the singleton asynchronous thread context, and which means that upon
321 * returning from both API methods, the client is not guaranteed that
322 * it has registered or unregistered successfully.
323 */
324
325 /* Keep the published item in circular ring buffer for
326 * efficient access to playback to new subscribers whose misses the
327 * data.
328 */
329
330 m_buffer.push_back(item);
331
332 std::size_t numOfElementToBeRemoved = 0;
333 if (m_buffer.size() > m_capacity) {
334 numOfElementToBeRemoved = m_buffer.size() - m_capacity;
335 }
336
337 while (numOfElementToBeRemoved > 0 && !m_buffer.empty()) {
338 m_buffer.erase(m_buffer.begin());
339 numOfElementToBeRemoved--;
340 }
341
342 for (auto &sub : m_subscribers) {
343 if (!m_filter_fn || m_filter_fn(sub.get(), item)) {
344 sub->notify(item, this);
345 }
346 }
347} // method publishInternal()

349template <typename T, template <class> class QueueType>
350void Dmn_Pub<T, QueueType>::registerSubscriber(std::shared_ptr<Dmn_Sub> sub) {
351 auto waitHandler = this->addExecTaskWithWait([this, sub]() -> void {
352 auto it = std::find(sub->m_pubs.begin(), sub->m_pubs.end(), this);
353 if (it != sub->m_pubs.end()) {
354 return;
355 }
356
357 sub->m_pubs.push_back(this);
358 m_subscribers.push_back(sub);
359
360 // resend the data items that the registered subscriber
361 // miss.
362 size_t numberOfItemsToBeSkipped = 0;
363 if (sub->m_replayQuantity > 0 &&
364 m_buffer.size() > static_cast<size_t>(sub->m_replayQuantity)) {
365 numberOfItemsToBeSkipped =
366 m_buffer.size() - static_cast<size_t>(sub->m_replayQuantity);
367 }
368
369 for (auto &item : m_buffer) {
370 if (numberOfItemsToBeSkipped > 0) {
371 numberOfItemsToBeSkipped--;
372 } else {
373 sub->notify(item, this);
374 }
375 }
376 });
377
378 waitHandler->wait();
379}
