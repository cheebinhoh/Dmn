326template <typename T>
327auto Dmn_BlockingQueue_Mt<T>::pop(size_t count, long timeout)
328 -> std::vector<T> {
329 std::vector<T> ret{};
330
331 assert(count > 0);
332
333 auto inflightTicket = this->enterInflightGate();
334
335 std::unique_lock<std::mutex> lock(m_mutex);
336
337 DMN_PROC_CLEANUP_PUSH(&Dmn_BlockingQueue_Mt<T>::cleanup_thunk_inflight,
338 &inflightTicket);
339
340 Dmn_Proc::testcancel();
341
342 // Wait until there are at least 'count' items OR a timeout occurs with at
343 // least one available item.
344 if (timeout > 0) {
345 if (m_not_empty_cond.wait_for(
346 lock, std::chrono::microseconds(timeout), [this, count] {
347 return m_queue.size() >= count || this->isInflightGuardClosed();
348 })) {
349 // do nothing and we have what we want
350 } else if (m_queue.empty()) {
351 // do nothing and fetch zero
352 } else {
353 // do nothing and fetch whatever we have
354 }
355 } else {
356 m_not_empty_cond.wait(lock, [this, count] {
357 return m_queue.size() >= count || this->isInflightGuardClosed();
358 });
359 }
360
361 // Collect up to 'count' items (moved out).
362 while (count > 0 && (!m_queue.empty())) {
363 ret.push_back(std::move_if_noexcept(m_queue.front()));
364 m_queue.pop_front();
365 ++m_pop_count;
366
367 count--;
368 }
369
370 // If queue became empty as a result of this pop, notify waitForEmpty()
371 // waiters.
372 bool empty = m_queue.empty();
373 lock.unlock();
374
375 if (empty) {
376 m_empty_cond.notify_all();
377 } else {
378 m_not_empty_cond.notify_one();
379 }
380
381 DMN_PROC_CLEANUP_POP(0);
382
383 return ret;
384}
