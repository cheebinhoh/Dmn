299template <typename T> auto Dmn_BlockingQueue<T>::waitForEmpty() -> uint64_t {

306 m_empty_cond.wait(lock, [this] {

316template <typename T>
317auto Dmn_BlockingQueue<T>::pop(size_t count, long timeout) -> std::vector<T> {
318 std::vector<T> ret{};
319
320 assert(count > 0);
321
322 auto inflightTicket = this->enterInflightGate();
323
324 std::unique_lock<std::mutex> lock(m_mutex);
325
326 DMN_PROC_CLEANUP_PUSH(&Dmn_BlockingQueue<T>::cleanup_thunk_inflight,
327 &inflightTicket);
328
329 Dmn_Proc::testcancel();
330
331 // Wait until there are at least 'count' items OR a timeout occurs with at
332 // least one available item.
333 if (timeout > 0) {
334 if (m_not_empty_cond.wait_for(
335 lock, std::chrono::microseconds(timeout), [this, count] {
336 return m_queue.size() >= count || this->isInflightGuardClosed();
337 })) {
338 // do nothing and we have what we want
339 } else if (m_queue.empty()) {
340 return {};
341 } else {
342 // do nothing and fetch whatever we have
343 }
344 } else {
345 m_not_empty_cond.wait(lock, [this, count] {
346 return m_queue.size() >= count || this->isInflightGuardClosed();
347 });
348 }
349
350 if (this->isInflightGuardClosed()) {
351 return ret;
352 }
353
354 // Collect up to 'count' items (moved out).
355 do {
356 ret.push_back(std::move_if_noexcept(m_queue.front()));
357 m_queue.pop_front();
358 ++m_pop_count;
359
360 count--;
361 } while (count > 0 && (!m_queue.empty()));
362
363 // If queue became empty as a result of this pop, notify waitForEmpty()
364 // waiters.
365 bool empty = m_queue.empty();
366 lock.unlock();
367
368 if (empty) {
369 m_empty_cond.notify_all();
370 } else {
371 m_not_empty_cond.notify_one();
372 }
373
374 DMN_PROC_CLEANUP_POP(0);
375
376 return ret;
377}

379template <typename T>
380auto Dmn_BlockingQueue<T>::popOptional(bool wait) -> std::optional<T> {
381 std::unique_lock<std::mutex> lock(m_mutex);
382
383 std::optional<T> ret{};
384
385 auto inflightTicket = this->enterInflightGate();
386
387 DMN_PROC_CLEANUP_PUSH(&Dmn_BlockingQueue<T>::cleanup_thunk_inflight,
388 &inflightTicket);
389
390 Dmn_Proc::testcancel();
391
392 if (m_queue.empty()) {
393 if (wait) {
394 // Block until an item is available. This wait is a cancellation point.
395 m_not_empty_cond.wait(lock, [this] {
396 return !m_queue.empty() || this->isInflightGuardClosed();
397 });
398
399 if (!this->isInflightGuardClosed()) {
400 ret = std::move_if_noexcept(m_queue.front());
401 m_queue.pop_front();
402 ++m_pop_count;
403 }
404 }
405 } else {
406 ret = std::move_if_noexcept(m_queue.front());
407 m_queue.pop_front();

412
413 lock.unlock();
414
415 // Notify waiters waiting for the queue to become empty.
416 if (empty) {
417 m_empty_cond.notify_all();
418 } else {
419 m_not_empty_cond.notify_one();
420 }
